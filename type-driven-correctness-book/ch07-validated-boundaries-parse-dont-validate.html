Type-Driven Correctness in Rust

验证边界 —— 解析，不要验证 🟡

你将学到： 如何在系统边界恰好验证一次数据，在专用类型中携带有效性证明，然后不再重新检查 —— 应用于 IPMI FRU 记录（扁平字节）、Redfish JSON（结构化文档）和 IPMI SEL 记录（多态二进制带嵌套分发），带有完整的端到端演练。

交叉引用： ch02（类型化命令）、ch06（量纲类型）、ch11（技巧 2 —— 密封 trait，技巧 3 —— #[non_exhaustive]，技巧 5 —— FromStr）、ch14（proptest）

问题：散弹枪验证

在典型代码中，验证分散在各处。每个接收数据的函数都重新检查它"以防万一"：

// C —— 验证分散在代码库中
int process_fru_data(uint8_t *data, int len) {
    if (data == NULL) return -1;          // 检查：非空
    if (len < 8) return -1;              // 检查：最小长度
    if (data[0] != 0x01) return -1;      // 检查：格式版本
    if (checksum(data, len) != 0) return -1; // 检查：校验和

    // ... 10 多个重复相同检查的函数 ...
}

这种模式（"散弹枪验证"）有两个问题：

  1. 冗余 —— 相同的检查出现在数十个地方
  2. 不完整 —— 在一个地方忘记一个检查，你就有了 bug

解析，不要验证

正确构造方法：在边界验证一次，然后在类型中携带有效性证明。

/// 来自线路的原始字节 —— 尚未验证。
#[derive(Debug)]
pub struct RawFruData(Vec<u8>);

案例研究：IPMI FRU 数据

#[derive(Debug)]
pub struct RawFruData(Vec<u8>);

/// 验证的 IPMI FRU 数据。只能通过 TryFrom 创建，
/// 它强制执行所有不变量。一旦你有了 ValidFru，
/// 所有数据都保证正确。
#[derive(Debug)]
pub struct ValidFru {
    format_version: u8,
    internal_area_offset: u8,
    chassis_area_offset: u8,
    board_area_offset: u8,
    product_area_offset: u8,
    data: Vec<u8>,
}

#[derive(Debug)]
pub enum FruError {
    TooShort { actual: usize, minimum: usize },
    BadFormatVersion(u8),
    ChecksumMismatch { expected: u8, actual: u8 },
    InvalidAreaOffset { area: &'static str, offset: u8 },
}

impl TryFrom<RawFruData> for ValidFru {
    type Error = FruError;

    fn try_from(raw: RawFruData) -> Result<Self, FruError> {
        let data = raw.0;

        if data.len() < 8 {
            return Err(FruError::TooShort { actual: data.len(), minimum: 8 });
        }
        if data[0] != 0x01 {
            return Err(FruError::BadFormatVersion(data[0]));
        }

        let checksum: u8 = data[..8].iter().fold(0u8, |acc, &b| acc.wrapping_add(b));
        if checksum != 0 {
            return Err(FruError::ChecksumMismatch { expected: 0, actual: checksum });
        }

        for (name, idx) in [("internal", 1), ("chassis", 2), ("board", 3), ("product", 4)] {
            let offset = data[idx];
            if offset != 0 && (offset as usize * 8) >= data.len() {
                return Err(FruError::InvalidAreaOffset { area: name, offset });
            }
        }

        Ok(ValidFru {
            format_version: data[0],
            internal_area_offset: data[1],
            chassis_area_offset: data[2],
            board_area_offset: data[3],
            product_area_offset: data[4],
            data,
        })
    }
}

impl ValidFru {
    pub fn board_area(&self) -> Option<&[u8]> {
        if self.board_area_offset == 0 { return None; }
        let start = self.board_area_offset as usize * 8;
        Some(&self.data[start..])
    }
}

任何接受 &ValidFru 的函数都知道数据格式良好。无需重新检查。

多态验证：IPMI SEL 记录

IPMI SEL 记录是多态的：字节的含义取决于记录类型和传感器类型。

步骤 1 —— 解析外框

pub struct RawSelRecord(pub [u8; 16]);

pub enum ValidSelRecord {
    SystemEvent(SystemEventRecord),
    OemTimestamped(OemTimestampedRecord),
    OemNonTimestamped(OemNonTimestampedRecord),
}

impl TryFrom<RawSelRecord> for ValidSelRecord {
    type Error = SelParseError;

    fn try_from(raw: RawSelRecord) -> Result<Self, SelParseError> {
        let d = &raw.0;
        let record_id = u16::from_le_bytes([d[0], d[1]]);

        match d[2] {
            0x02 => Ok(ValidSelRecord::SystemEvent(parse_system_event(record_id, d)?)),
            0xC0..=0xDF => Ok(ValidSelRecord::OemTimestamped(OemTimestampedRecord {
                record_id,
                timestamp: u32::from_le_bytes([d[3], d[4], d[5], d[6]]),
                manufacturer_id: [d[7], d[8], d[9]],
                oem_data: [d[10], d[11], d[12], d[13], d[14], d[15]],
            })),
            0xE0..=0xFF => Ok(ValidSelRecord::OemNonTimestamped(OemNonTimestampedRecord {
                record_id,
                oem_data: [d[3], d[4], d[5], d[6], d[7], d[8], d[9], d[10], d[11], d[12], d[13], d[14], d[15]],
            })),
            other => Err(SelParseError::UnknownRecordType(other)),
        }
    }
}

步骤 2 —— 类型化事件枚举

#[non_exhaustive]
#[derive(Debug, Clone, Copy, PartialEq)]
pub enum SensorType {
    Temperature, Voltage, Current, Fan, PhysicalSecurity,
    Processor, PowerSupply, Memory, SystemEvent, Watchdog2,
}

#[derive(Debug)]
pub enum TypedEvent {
    Threshold(ThresholdEvent),
    SensorSpecific(SensorSpecificEvent),
    Discrete { offset: u8, event_data: [u8; 3] },
}

#[derive(Debug)]
pub enum SensorSpecificEvent {
    Temperature(TempEvent),
    Voltage(VoltageEvent),
    Fan(FanEvent),
    Processor(ProcessorEvent),
    PowerSupply(PowerSupplyEvent),
    Memory(MemoryEvent),
    PhysicalSecurity(PhysicalSecurityEvent),
    Watchdog(WatchdogEvent),
}

#[derive(Debug, Clone, Copy, PartialEq)]
pub enum MemoryEvent {
    CorrectableEcc, UncorrectableEcc, Parity, CriticalOvertemperature,

    PresenceDetected, Failure, PredictiveFailure, InputLost,

步骤 3 —— 消费类型化 SEL 记录

fn should_alert(record: &ValidSelRecord) -> bool {
    match record {
        ValidSelRecord::SystemEvent(sys) => match &sys.event {
            TypedEvent::Threshold(t) => {
                matches!(t.crossing,
                    ThresholdCrossing::UpperCriticalLow | ThresholdCrossing::UpperCriticalHigh
                    | ThresholdCrossing::LowerCriticalLow | ThresholdCrossing::LowerCriticalHigh
                    | ThresholdCrossing::UpperNonRecoverableLow | ThresholdCrossing::UpperNonRecoverableHigh
                )
            }
            TypedEvent::SensorSpecific(ss) => match ss {
                SensorSpecificEvent::Memory(m) => matches!(m,
                    MemoryEvent::UncorrectableEcc | MemoryEvent::Parity | MemoryEvent::CriticalOvertemperature
                ),
                SensorSpecificEvent::PowerSupply(p) => matches!(p,
                    PowerSupplyEvent::Failure | PowerSupplyEvent::InputLost
                ),
                _ => false,
            },
            TypedEvent::Discrete { .. } => false,
        },
        ValidSelRecord::OemTimestamped(_) | ValidSelRecord::OemNonTimestamped(_) => false,
    }
}

关键要点

  1. 在边界解析一次 —— TryFrom<Raw> for Valid 是验证发生的地方。
  2. 多态数据变成嵌套枚举 —— 编译器强制执行穷举处理。
  3. 下游代码从不重新验证 —— ValidFru 和 ValidSelRecord 保证正确性。
  4. 从解析到 Redfish 健康的消费者管道 —— SEL 记录通过解析 → 分类 → 线性化 → 聚合，生成 TypedSelSummary 用于 ch18 的健康汇总。
