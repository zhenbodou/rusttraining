C# 开发者选择 Rust 的理由

你将学到： 为什么 Rust 对 C# 开发者很重要——托管代码和原生代码之间的性能差距， Rust 如何在编译时消除空引用异常和隐藏的控制流， 以及 Rust 补充或替代 C# 的关键场景。

难度： 🟢 初学者

无需运行时开销的性能

// C# - 出色的生产力，运行时开销
public class DataProcessor
{
    private List<int> data = new List<int>();

    public void ProcessLargeDataset()
    {
        // 分配触发 GC
        for (int i = 0; i < 10_000_000; i++)
        {
            data.Add(i * 2); // GC 压力
        }
        // 处理期间不可预测的 GC 暂停
    }
}
// 运行时：可变（50-200ms，由于 GC）
// 内存：~80MB（包括 GC 开销）
// 可预测性：低（GC 暂停）

#![allow(unused)]
fn main() {
// Rust - 同样的表达能力，零运行时开销
struct DataProcessor {
    data: Vec<i32>,
}

impl DataProcessor {
    fn process_large_dataset(&mut self) {
        // 零成本抽象
        for i in 0..10_000_000 {
            self.data.push(i * 2); // 无 GC 压力
        }
        // 确定性性能
    }
}
// 运行时：一致（~30ms）
// 内存：~40MB（精确分配）
// 可预测性：高（无 GC）
}

无需运行时检查的内存安全

// C# - 运行时安全，有开销
public class RuntimeCheckedOperations
{
    public string? ProcessArray(int[] array)
    {
        // 每次访问都有运行时边界检查
        if (array.Length > 0)
        {
            return array[0].ToString(); // 安全——int 是值类型，永不为 null
        }
        return null; // 可空返回（C# 8+ 可空引用类型）
    }

    public void ProcessConcurrently()
    {
        var list = new List<int>();

        // 可能发生数据竞争，需要仔细锁定
        Parallel.For(0, 1000, i =>
        {
            lock (list) // 运行时开销
            {
                list.Add(i);
            }
        });
    }
}

#![allow(unused)]
fn main() {
// Rust - 编译时安全，零运行时成本
struct SafeOperations;

impl SafeOperations {
    // 编译时空安全，无需运行时检查
    fn process_array(array: &[i32]) -> Option<String> {
        array.first().map(|x| x.to_string())
        // 不可能有空引用
        // 当可证明安全时，边界检查会被优化掉
    }

    fn process_concurrently() {
        use std::sync::{Arc, Mutex};
        use std::thread;

        let data = Arc::new(Mutex::new(Vec::new()));

        // 编译时防止数据竞争
        let handles: Vec<_> = (0..1000).map(|i| {
            let data = Arc::clone(&data);
            thread::spawn(move || {
                data.lock().unwrap().push(i);
            })
        }).collect();

        for handle in handles {
            handle.join().unwrap();
        }
    }
}
}

Rust 解决的常见 C# 痛点

1. 十亿美元的错误：空引用

// C# - 空引用异常是运行时炸弹
public class UserService
{
    public string GetUserDisplayName(User user)
    {
        // 这些都可能抛出 NullReferenceException
        return user.Profile.DisplayName.ToUpper();
        //     ^^^^^ ^^^^^^^ ^^^^^^^^^^^ ^^^^^^^
        //     可能在运行时为 null
    }

    // 即使使用可空引用类型（C# 8+）
    public string GetDisplayName(User? user)
    {
        return user?.Profile?.DisplayName?.ToUpper() ?? "Unknown";
        // 运行时仍可能出现 null
    }
}

#![allow(unused)]
fn main() {
// Rust - 编译时保证的空安全
struct UserService;

impl UserService {
    fn get_user_display_name(user: &User) -> Option<String> {
        user.profile.as_ref()?
            .display_name.as_ref()
            .map(|name| name.to_uppercase())
        // 编译器强制你处理 None 情况
        // 不可能有空指针异常
    }

    fn get_display_name_safe(user: Option<&User>) -> String {
        user.and_then(|u| u.profile.as_ref())
            .and_then(|p| p.display_name.as_ref())
            .map(|name| name.to_uppercase())
            .unwrap_or_else(|| "Unknown".to_string())
        // 显式处理，无意外
    }
}
}

2. 隐藏异常和控制流

// C# - 异常可能从任何地方抛出
public async Task<UserData> GetUserDataAsync(int userId)
{
    // 每一个都可能抛出不同的异常
    var user = await userRepository.GetAsync(userId);        // SqlException
    var permissions = await permissionService.GetAsync(user); // HttpRequestException
    var preferences = await preferenceService.GetAsync(user); // TimeoutException

    return new UserData(user, permissions, preferences);
    // 调用者不知道会抛出什么异常
}

#![allow(unused)]
fn main() {
// Rust - 所有错误在函数签名中显式声明
#[derive(Debug)]
enum UserDataError {
    DatabaseError(String),
    NetworkError(String),
    Timeout,
    UserNotFound(i32),
}

async fn get_user_data(user_id: i32) -> Result<UserData, UserDataError> {
    // 所有错误都是显式处理的
    let user = user_repository.get(user_id).await
        .map_err(UserDataError::DatabaseError)?;

    let permissions = permission_service.get(&user).await
        .map_err(UserDataError::NetworkError)?;

    let preferences = preference_service.get(&user).await
        .map_err(|_| UserDataError::Timeout)?;

    Ok(UserData::new(user, permissions, preferences))
    // 调用者确切知道可能出现什么错误
}
}

3. 正确性：类型系统作为证明引擎

Rust 的类型系统在编译时捕获了 C# 只能在运行时捕获（或根本无法捕获）的整类逻辑错误。

ADT 与密封类变通方案

// C# —— 判别联合需要密封类样板代码。
// 仅当没有 _ 通配符时，编译器才会警告缺失的情况（CS8524）。
// 实际上，大多数 C# 代码使用 _ 作为默认值，这会抑制警告。
public abstract record Shape;
public sealed record Circle(double Radius)   : Shape;
public sealed record Rectangle(double W, double H) : Shape;
public sealed record Triangle(double A, double B, double C) : Shape;

public static double Area(Shape shape) => shape switch
{
    Circle c    => Math.PI * c.Radius * c.Radius,
    Rectangle r => r.W * r.H,
    // 忘记 Triangle？_ 通配符抑制了任何编译器警告。
    _           => throw new ArgumentException("未知形状")
};
// 六个月后添加一个新变体——_ 模式隐藏了缺失的情况。
// 没有编译器警告告诉你需要更新 47 个 switch 表达式。

#![allow(unused)]
fn main() {
// Rust —— ADT + 穷尽匹配 = 编译时证明
enum Shape {
    Circle { radius: f64 },
    Rectangle { w: f64, h: f64 },
    Triangle { a: f64, b: f64, c: f64 },
}

fn area(shape: &Shape) -> f64 {
    match shape {
        Shape::Circle { radius }    => std::f64::consts::PI * radius * radius,
        Shape::Rectangle { w, h }   => w * h,
        // 忘记 Triangle？错误：模式不穷尽
        Shape::Triangle { a, b, c } => {
            let s = (a + b + c) / 2.0;
            (s * (s - a) * (s - b) * (s - c)).sqrt()
        }
    }
}
// 添加新变体 → 编译器显示每个需要更新的 match。
}

默认不可变性与选择性不可变性

// C# —— 默认一切皆可变
public class Config
{
    public string Host { get; set; }   // 默认可变
    public int Port { get; set; }
}

// "readonly" 和 "record" 有帮助，但不能防止深层变异：
public record ServerConfig(string Host, int Port, List<string> AllowedOrigins);

var config = new ServerConfig("localhost", 8080, new List<string> { "*.example.com" });
// Record 是"不可变的"，但引用类型字段 NOT：
config.AllowedOrigins.Add("*.evil.com"); // 编译并通过并变异！← bug
// 编译器不给你任何警告。

#![allow(unused)]
fn main() {
// Rust —— 默认不可变，变异是显式且可见的
struct Config {
    host: String,
    port: u16,
    allowed_origins: Vec<String>,
}

let config = Config {
    host: "localhost".into(),
    port: 8080,
    allowed_origins: vec!["*.example.com".into()],
};

// config.allowed_origins.push("*.evil.com".into()); // 错误：无法可变借用

// 变异需要显式选择加入：
let mut config = config;
config.allowed_origins.push("*.safe.com".into()); // 好——可见可变

// 签名中的 "mut" 告诉每个读者："此函数修改数据"
fn add_origin(config: &mut Config, origin: String) {
    config.allowed_origins.push(origin);
}
}

函数式编程：一等公民 vs 事后想法

// C# —— FP 强加；LINQ 有表现力但语言与你作对
public IEnumerable<Order> GetHighValueOrders(IEnumerable<Order> orders)
{
    return orders
        .Where(o => o.Total > 1000)   // Func<Order, bool> —— 堆分配委托
        .Select(o => new OrderSummary  // 匿名类型或额外类
        {
            Id = o.Id,
            Total = o.Total
        })
        .OrderByDescending(o => o.Total);
    // 结果上没有穷尽匹配
    // null 可能在管道中任何地方潜入
    // 无法强制纯度——任何 lambda 都可能有副作用
}

#![allow(unused)]
fn main() {
// Rust —— FP 是一等公民
fn get_high_value_orders(orders: &[Order]) -> Vec<OrderSummary> {
    orders.iter()
        .filter(|o| o.total > 1000)      // 零成本闭包，无堆分配
        .map(|o| OrderSummary {           // 类型检查结构体
            id: o.id,
            total: o.total,
        })
        .sorted_by(|a, b| b.total.cmp(&a.total)) // itertools
        .collect()
    // 管道中没有任何 null
    // 闭包是单态化的——与手写循环相比零开销
    // 强制纯度：&[Order] 意味着函数不能修改 orders
}
}

继承：理论上优雅，实践中脆弱

// C# —— 脆弱基类问题
public class Animal
{
    public virtual string Speak() => "...";
    public void Greet() => Console.WriteLine($"I say: {Speak()}");
}

public class Dog : Animal
{
    public override string Speak() => "Woof!";
}

public class RobotDog : Dog
{
    // Greet() 调用哪个 Speak()？如果 Dog 改变会怎样？
    // 接口 + 默认方法的菱形问题
    // 紧密耦合：改变 Animal 可能静默破坏 RobotDog
}

// 常见 C# 反模式：
// - 有 20 个虚方法的上帝基类
// - 没人能理解的深层层次结构（5+ 层）
// - 创建隐藏耦合的 "protected" 字段
// - 基类更改静默改变派生行为

#![allow(unused)]
fn main() {
// Rust —— 组合优于继承，由语言强制
trait Speaker {
    fn speak(&self) -> &str;
}

trait Greeter: Speaker {
    fn greet(&self) {
        println!("I say: {}", self.speak());
    }
}

struct Dog;
impl Speaker for Dog {
    fn speak(&self) -> &str { "Woof!" }
}
impl Greeter for Dog {} // 使用默认 greet()

struct RobotDog {
    voice: String, // 组合：拥有自己的数据
}
impl Speaker for RobotDog {
    fn speak(&self) -> &str { &self.voice }
}
impl Greeter for RobotDog {} // 清晰、显式的行为

// 没有脆弱基类问题——根本没有基类
// 没有隐藏耦合——trait 是显式契约
// 没有菱形问题——trait 一致性规则防止歧义
// 向 Speaker 添加方法？编译器告诉你在哪里实现它。
}

关键见解：在 C# 中，正确性是一种纪律——你希望开发者 遵循约定、编写测试、在代码审查中捕获边界情况。 在 Rust 中，正确性是类型系统的属性——整类错误 （空指针解引用、忘记的变体、意外变异、数据竞争）在结构上是不可能的。

4. 由于 GC 导致的不可预测性能

// C# - GC 可能随时暂停
public class HighFrequencyTrader
{
    private List<Trade> trades = new List<Trade>();

    public void ProcessMarketData(MarketTick tick)
    {
        // 分配可能在最关键时刻触发 GC
        var analysis = new MarketAnalysis(tick);
        trades.Add(new Trade(analysis.Signal, tick.Price));

        // GC 可能在关键市场时刻暂停这里
        // 暂停持续时间：1-100ms，取决于堆大小
    }
}

#![allow(unused)]
fn main() {
// Rust - 可预测的确定性性能
struct HighFrequencyTrader {
    trades: Vec<Trade>,
}

impl HighFrequencyTrader {
    fn process_market_data(&mut self, tick: MarketTick) {
        // 零分配，可预测的性能
        let analysis = MarketAnalysis::from(tick);
        self.trades.push(Trade::new(analysis.signal(), tick.price));

        // 无 GC 暂停，一致的亚微秒级延迟
        // 类型系统保证的性能
    }
}
}

何时选择 Rust 而非 C#

✅ 选择 Rust 当：

  • 正确性很重要：状态机、协议实现、财务逻辑——遗漏的情况是生产事件，而不是测试失败
  • 性能至关重要：实时系统、高频交易、游戏引擎
  • 内存使用很重要：嵌入式系统、云成本、移动应用
  • 需要可预测性：医疗设备、汽车、金融系统
  • 安全至关重要：密码学、网络安全、系统级代码
  • 长期运行的服务：GC 暂停会导致问题
  • 资源受限环境：IoT、边缘计算
  • 系统编程：CLI 工具、数据库、Web 服务器、操作系统

✅ 继续使用 C# 当：

  • 快速应用开发：业务应用、CRUD 应用
  • 大型现有代码库：迁移成本过高时
  • 团队专业知识：Rust 学习曲线不值得时
  • 企业集成：重度 .NET Framework/Windows 依赖
  • GUI 应用：WPF、WinUI、Blazor 生态系统
  • 上市时间：开发速度胜过性能时

🔄 考虑两者（混合方法）：

  • Rust 中的性能关键组件：通过 P/Invoke 从 C# 调用
  • C# 中的业务逻辑：熟悉、高效的开发
  • 渐进式迁移：在 Rust 中开始新服务

现实世界影响：为什么公司选择 Rust

Dropbox：存储基础设施

  • 之前（Python）：高 CPU 使用率，内存开销
  • 之后（Rust）：10 倍性能提升，50% 内存减少
  • 结果：节省数百万基础设施成本

Discord：语音/视频后端

  • 之前（Go）：GC 暂停导致音频丢失
  • 之后（Rust）：一致的低延迟性能
  • 结果：更好的用户体验，降低服务器成本

Microsoft：Windows 组件

  • Windows 中的 Rust：文件系统、网络栈组件
  • 优势：内存安全，无性能成本
  • 影响：更少的安全漏洞，相同的性能

为什么这对 C# 开发者很重要：

  1. 互补技能：Rust 和 C# 解决不同的问题
  2. 职业成长：系统编程专业知识越来越有价值
  3. 性能理解：学习零成本抽象
  4. 安全思维：将所有权思维应用于任何语言
  5. 云成本：性能直接影响基础设施支出

语言哲学比较

C# 哲学

  • 生产力优先：丰富的工具、广泛的框架、"成功之坑"
  • 托管运行时：垃圾回收自动处理内存
  • 面向企业：强类型与反射、广泛的标准库
  • 面向对象：类、继承、接口作为主要抽象

Rust 哲学

  • 无牺牲的性能：零成本抽象，无运行时开销
  • 内存安全：编译时保证防止崩溃和安全漏洞
  • 系统编程：直接硬件访问与高级抽象
  • 函数式 + 系统：默认不可变、基于所有权的资源管理
