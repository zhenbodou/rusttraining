using System;

    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Hello, World!");
        }
    }
}

Rust Hello World

// main.rs
fn main() {
    println!("Hello, World!");
}

C# 开发者的关键区别

  1. 不需要类 —— 函数可以存在于顶层
  2. 没有命名空间 —— 使用模块系统代替
  3. println! 是宏 —— 注意 !
  4. println! 后没有分号 —— 表达式 vs 语句
  5. 没有显式返回类型 —— main 返回 ()（单元类型）

读取输入和 CLI 参数

每个 C# 开发者都知道 Console.ReadLine()。以下是如何在 Rust 中处理用户输入、环境变量和命令行参数。

控制台输入

// C# —— 读取用户输入
Console.Write("Enter your name: ");
string? name = Console.ReadLine();  // .NET 6+ 返回 string?
Console.WriteLine($"Hello, {name}!");

// 解析输入
Console.Write("Enter a number: ");
if (int.TryParse(Console.ReadLine(), out int number))
{
    Console.WriteLine($"You entered: {number}");
}
else
{
    Console.WriteLine("That's not a valid number.");
}

use std::io::{self, Write};

fn main() {
    // 读取一行输入
    print!("Enter your name: ");
    io::stdout().flush().unwrap(); // 刷新，因为 print! 不会自动刷新

    let mut name = String::new();
    io::stdin().read_line(&mut name).expect("Failed to read line");
    let name = name.trim(); // 移除尾随换行符
    println!("Hello, {name}!");

    // 解析输入
    print!("Enter a number: ");
    io::stdout().flush().unwrap();

    let mut input = String::new();
    io::stdin().read_line(&mut input).expect("Failed to read");
    match input.trim().parse::<i32>() {
        Ok(number) => println!("You entered: {number}"),
        Err(_)     => println!("That's not a valid number."),
    }
}

命令行参数

// C# —— 读取 CLI 参数
static void Main(string[] args)
{
    if (args.Length < 1)
    {
        Console.WriteLine("Usage: program <filename>");
        return;
    }
    string filename = args[0];
    Console.WriteLine($"Processing {filename}");
}

use std::env;

fn main() {
    let args: Vec<String> = env::args().collect();
    //  args[0] = 程序名（类似 C# 的程序集名称）
    //  args[1..] = 实际参数

    if args.len() < 2 {
        eprintln!("Usage: {} <filename>", args[0]); // eprintln! → stderr
        std::process::exit(1);
    }
    let filename = &args[1];
    println!("Processing {filename}");
}

使用 clap 的生产级 CLI 应用

对于超出简单参数解析的任何内容，请使用 clap crate——它是 Rust 中等价于 System.CommandLine 或 CommandLineParser 等库的工具。

# Cargo.toml
[dependencies]
clap = { version = "4", features = ["derive"] }

use clap::Parser;

/// 一个简单的文件处理器——此文档注释成为帮助文本
#[derive(Parser, Debug)]
#[command(name = "processor", version, about)]
struct Args {
    /// 要处理的输入文件
    #[arg(short, long)]
    input: String,

    /// 输出文件（默认为 stdout）
    #[arg(short, long)]
    output: Option<String>,

    /// 启用详细日志
    #[arg(short, long, default_value_t = false)]
    verbose: bool,

    /// 工作线程数
    #[arg(short = 'j', long, default_value_t = 4)]
    threads: usize,
}

fn main() {
    let args = Args::parse(); // 自动解析、验证、生成 --help

    if args.verbose {
        println!("Input:   {}", args.input);
        println!("Output:  {:?}", args.output);
        println!("Threads: {}", args.threads);
    }

    // 使用 args.input、args.output 等
}
# 自动生成的帮助：
$ processor --help
A simple file processor

Usage: processor [OPTIONS] --input <INPUT>

Options:
  -i, --input <INPUT>      Input file to process
  -o, --output <OUTPUT>    Output file (defaults to stdout)
  -v, --verbose            Enable verbose logging
  -j, --threads <THREADS>  Number of worker threads [default: 4]
  -h, --help               Print help
  -V, --version            Print version

// C# 中使用 System.CommandLine 的等价代码（更多样板代码）：
var inputOption = new Option<string>("--input", "Input file") { IsRequired = true };
var verboseOption = new Option<bool>("--verbose", "Enable verbose logging");
var rootCommand = new RootCommand("A simple file processor");
rootCommand.AddOption(inputOption);
rootCommand.AddOption(verboseOption);
rootCommand.SetHandler((input, verbose) => { /* ... */ }, inputOption, verboseOption);
await rootCommand.InvokeAsync(args);
// clap 的 derive 宏方法更简洁且类型安全
