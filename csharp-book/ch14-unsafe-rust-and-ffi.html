Rust for C# Programmers

Unsafe Rust

你将学到： unsafe 允许什么（原始指针、FFI、未检查的强制转换），安全包装器模式， C# P/Invoke vs Rust FFI 用于调用原生代码，以及 unsafe 块的安全检查清单。

难度： 🔴 高级

Unsafe Rust 允许你执行借用检查器无法验证的操作。少用并带有清晰的文档。

高级覆盖：对于 unsafe 代码上的安全抽象模式（arena 分配器、无锁结构、自定义 vtable），请参阅 Rust Patterns。

何时需要 Unsafe

#![allow(unused)]
fn main() {
// 1. 解引用原始指针
let mut value = 42;
let ptr = &mut value as *mut i32;
// 安全：ptr 指向有效、活动的局部变量。
unsafe {
    *ptr = 100; // 必须在 unsafe 块中
}

// 2. 调用 unsafe 函数
unsafe fn dangerous() {
    // 需要调用者维护不变量的内部实现
}

// 安全：此示例函数无需维护不变量。
unsafe {
    dangerous(); // 调用者承担责任
}

// 3. 访问可变静态变量
static mut COUNTER: u32 = 0;
// 安全：单线程上下文；无对 COUNTER 的并发访问。
unsafe {
    COUNTER += 1; // 不是线程安全——调用者必须确保同步
}

// 4. 实现 unsafe traits
unsafe trait UnsafeTrait {
    fn do_something(&self);
}
}

C# 比较：unsafe 关键字

// C# unsafe - 类似概念，不同范围
unsafe void UnsafeExample()
{
    int value = 42;
    int* ptr = &value;
    *ptr = 100;

    // C# unsafe 是关于指针算术
    // Rust unsafe 是关于所有权/借用规则放松
}

// C# fixed - 固定托管对象
unsafe void PinnedExample()
{
    byte[] buffer = new byte[100];
    fixed (byte* ptr = buffer)
    {
        // ptr 仅在此块内有效
    }
}

安全包装器

#![allow(unused)]
fn main() {
/// 关键模式：将 unsafe 代码包装在安全 API 中
pub struct SafeBuffer {
    data: Vec<u8>,
}

impl SafeBuffer {
    pub fn new(size: usize) -> Self {
        SafeBuffer { data: vec![0; size] }
    }

    /// 安全 API——边界检查访问
    pub fn get(&self, index: usize) -> Option<u8> {
        self.data.get(index).copied()
    }

    /// 快速未检查访问——unsafe 但用边界检查安全包装
    pub fn get_unchecked_safe(&self, index: usize) -> Option<u8> {
        if index < self.data.len() {
            // 安全：我们刚检查了 index 在边界内
            Some(unsafe { *self.data.get_unchecked(index) })
        } else {
            None
        }
    }
}
}

Rust 库（编译为 cdylib）

#![allow(unused)]
fn main() {
// src/lib.rs
#[no_mangle]
pub extern "C" fn add_numbers(a: i32, b: i32) -> i32 {
    a + b
}

#[no_mangle]
pub extern "C" fn process_string(input: *const std::os::raw::c_char) -> i32 {
    // 安全：input 非空（内部检查）并由调用者假定为 null 终止。
    let c_str = unsafe {
        if input.is_null() {
            return -1;
        }
        std::ffi::CStr::from_ptr(input)
    };

    match c_str.to_str() {
        Ok(s) => s.len() as i32,
        Err(_) => -1,
    }
}
}

C# 消费者（P/Invoke）

using System.Runtime.InteropServices;

public static class RustInterop
{
    [DllImport("my_rust_lib", CallingConvention = CallingConvention.Cdecl)]
    public static extern int add_numbers(int a, int b);

    [DllImport("my_rust_lib", CallingConvention = CallingConvention.Cdecl)]
    public static extern int process_string(
        [MarshalAs(UnmanagedType.LPUTF8Str)] string input);
}

// 用法
int sum = RustInterop.add_numbers(5, 3);  // 8
int len = RustInterop.process_string("Hello from C#!");  // 15

FFI 安全检查清单

当向 C# 暴露 Rust 函数时，这些规则防止最常见的 bug：

  1. 总是使用 extern "C" —— 没有它，Rust 使用自己的（不稳定的）调用约定。C# P/Invoke 期望 C ABI。

  2. #[no_mangle] —— 防止 Rust 编译器混淆函数名。没有它，C# 找不到符号。

  3. 永远不要让 panic 跨越 FFI 边界 —— Rust panic 展开到 C# 中是未定义行为。在 FFI 入口点捕获 panic：

    #![allow(unused)]
fn main() {
#[no_mangle]
pub extern "C" fn safe_ffi_function() -> i32 {
    match std::panic::catch_unwind(|| {
        // 实际逻辑在这里
        42
    }) {
        Ok(result) => result,
        Err(_) => -1,  // 返回错误代码而不是 panic 到 C# 中
    }
}
}

  4. 不透明 vs 透明结构体 —— 如果 C# 只持有指针（不透明句柄），不需要 #[repr(C)]。如果 C# 通过 StructLayout 读取结构体字段，你必须使用 #[repr(C)]：

    #![allow(unused)]
fn main() {
// 不透明——C# 只持有 IntPtr。不需要 #[repr(C)]。
pub struct Connection { /* 仅 Rust 字段 */ }

// 透明——C# marshal 字段直接。必须使用 #[repr(C)]。
#[repr(C)]
pub struct Point { pub x: f64, pub y: f64 }
}

  5. 空指针检查 —— 解引用前总是验证指针。C# 可以传递 IntPtr.Zero。

  6. 字符串编码 —— C# 内部使用 UTF-16。MarshalAs(UnmanagedType.LPUTF8Str) 转换为 Rust 的 CStr 的 UTF-8。显式记录此契约。

端到端示例：带生命周期管理的不透明句柄

这个模式在生产中常见：Rust 拥有对象，C# 持有不透明句柄，显式的 create/destroy 函数管理生命周期。

Rust 端（src/lib.rs）：

#![allow(unused)]
fn main() {
use std::ffi::{c_char, CStr};

pub struct ImageProcessor {
    width: u32,
    height: u32,
    pixels: Vec<u8>,
}

/// 创建新处理器。如果维度无效则返回 null。
#[no_mangle]
pub extern "C" fn processor_new(width: u32, height: u32) -> *mut ImageProcessor {
    if width == 0 || height == 0 {
        return std::ptr::null_mut();
    }
    let proc = ImageProcessor {
        width,
        height,
        pixels: vec![0u8; (width * height * 4) as usize],
    };
    Box::into_raw(Box::new(proc)) // 在堆上分配，返回原始指针
}

/// 应用灰度滤镜。成功返回 0，空指针返回 -1。
#[no_mangle]
pub extern "C" fn processor_grayscale(ptr: *mut ImageProcessor) -> i32 {
    // 安全：ptr 由 Box::into_raw 创建（非空），仍然有效。
    let proc = match unsafe { ptr.as_mut() } {
        Some(p) => p,
        None => return -1,
    };
    for chunk in proc.pixels.chunks_exact_mut(4) {
        let gray = (0.299 * chunk[0] as f64
                  + 0.587 * chunk[1] as f64
                  + 0.114 * chunk[2] as f64) as u8;
        chunk[0] = gray;
        chunk[1] = gray;
        chunk[2] = gray;
    }
    0
}

/// 销毁处理器。可以用 null 安全调用。
#[no_mangle]
pub extern "C" fn processor_free(ptr: *mut ImageProcessor) {
    if !ptr.is_null() {
        // 安全：ptr 由 processor_new 通过 Box::into_raw 创建
        unsafe { drop(Box::from_raw(ptr)); }
    }
}
}

C# 端：

using System.Runtime.InteropServices;

public sealed class ImageProcessor : IDisposable
{
    [DllImport("image_rust", CallingConvention = CallingConvention.Cdecl)]
    private static extern IntPtr processor_new(uint width, uint height);

    [DllImport("image_rust", CallingConvention = CallingConvention.Cdecl)]
    private static extern int processor_grayscale(IntPtr ptr);

    [DllImport("image_rust", CallingConvention = CallingConvention.Cdecl)]
    private static extern void processor_free(IntPtr ptr);

    private IntPtr _handle;

    public ImageProcessor(uint width, uint height)
    {
        _handle = processor_new(width, height);
        if (_handle == IntPtr.Zero)
            throw new ArgumentException("Invalid dimensions");
    }

    public void Grayscale()
    {
        if (processor_grayscale(_handle) != 0)
            throw new InvalidOperationException("Processor is null");
    }

    public void Dispose()
    {
        if (_handle != IntPtr.Zero)
        {
            processor_free(_handle);
            _handle = IntPtr.Zero;
        }
    }
}

// 用法——IDisposable 确保 Rust 内存被释放
using var proc = new ImageProcessor(1920, 1080);
proc.Grayscale();
// proc.Dispose() 自动调用 → processor_free() → Rust drop Vec

关键见解：这等效于 C# 的 SafeHandle 模式。Rust 的 Box::into_raw / Box::from_raw 跨 FFI 边界转移所有权，C# IDisposable 包装器确保清理。

练习

🏋️ 练习：原始指针的安全包装器（点击展开）

你从 C 库接收原始指针。编写安全 Rust 包装器：

#![allow(unused)]
fn main() {
// 模拟 C API
extern "C" {
    fn lib_create_buffer(size: usize) -> *mut u8;
    fn lib_free_buffer(ptr: *mut u8);
}
}

要求：

  1. 创建包装原始指针的 SafeBuffer 结构体
  2. 实现 Drop 调用 lib_free_buffer
  3. 通过 as_slice() 提供安全 &[u8] 视图
  4. 如果指针为 null 则确保 SafeBuffer::new() 返回 None
🔑 解决方案 
struct SafeBuffer {
    ptr: *mut u8,
    len: usize,
}

impl SafeBuffer {
    fn new(size: usize) -> Option<Self> {
        // 安全：lib_create_buffer 返回有效指针或 null（下面检查）。
        let ptr = unsafe { lib_create_buffer(size) };
        if ptr.is_null() {
            None
        } else {
            Some(SafeBuffer { ptr, len: size })
        }
    }

    fn as_slice(&self) -> &[u8] {
        // 安全：ptr 非空（在 new() 中检查），len 是
        // 分配的大小，我们持有独占所有权。
        unsafe { std::slice::from_raw_parts(self.ptr, self.len) }
    }
}

impl Drop for SafeBuffer {
    fn drop(&mut self) {
        // 安全：ptr 由 lib_create_buffer 分配
        unsafe { lib_free_buffer(self.ptr); }
    }
}

// 用法：所有 unsafe 包含在 SafeBuffer 中
fn process(buf: &SafeBuffer) {
    let data = buf.as_slice(); // 完全安全的 API
    println!("First byte: {}", data[0]);
}

关键模式：将 unsafe 封装在带 // SAFETY: 注释的小模块中。暴露 100% 安全的公共 API。这就是 Rust 标准库的工作方式——Vec、String、HashMap 内部都包含 unsafe 但呈现安全接口。
