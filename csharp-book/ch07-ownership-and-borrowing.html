Rust for C# Programmers

理解所有权

你将学到： Rust 的所有权系统——为什么 let s2 = s1 使 s1 失效（与 C# 引用复制不同）， 三个所有权规则，Copy vs Move 类型，使用 & 和 &mut 借用， 以及借用检查器如何替代垃圾回收。

难度： 🟡 中级

所有权是 Rust 最独特的特性，也是 C# 开发者最大的概念转变。让我们逐步进行。

C# 内存模型（回顾）

// C# - 自动内存管理
public void ProcessData()
{
    var data = new List<int> { 1, 2, 3, 4, 5 };
    ProcessList(data);
    // data 在这里仍然可以访问
    Console.WriteLine(data.Count);  // 工作正常

    // GC 将在没有引用剩余时清理
}

public void ProcessList(List<int> list)
{
    list.Add(6);  // 修改原始列表
}

Rust 所有权规则

  1. 每个值恰好有一个所有者（除非你选择使用 Rc<T>/Arc<T> 共享所有权——见 智能指针）
  2. 当所有者超出作用域时，值被丢弃（确定性清理——见 Drop）
  3. 所有权可以被转移（移动）
#![allow(unused)]
fn main() {
// Rust - 显式所有权管理
fn process_data() {
    let data = vec![1, 2, 3, 4, 5];  // data 拥有向量
    process_list(data);              // 所有权移动到函数
    // println!("{:?}", data);       // ❌ 错误：data 不再在这里拥有
}

fn process_list(mut list: Vec<i32>) {  // list 现在拥有向量
    list.push(6);
    // list 在函数结束时在这里被丢弃
}
}

C# 开发者的"移动"理解

// C# - 引用被复制，对象保持原位
// （只有引用类型——类——这样工作；
//  C# 值类型如 struct 行为不同）
var original = new List<int> { 1, 2, 3 };
var reference = original;  // 两个变量指向同一个对象
original.Add(4);
Console.WriteLine(reference.Count);  // 4 - 同一个对象

#![allow(unused)]
fn main() {
// Rust - 所有权被转移
let original = vec![1, 2, 3];
let moved = original;       // 所有权转移
// println!("{:?}", original);  // ❌ 错误：original 不再拥有数据
println!("{:?}", moved);    // ✅ 工作：moved 现在拥有数据
}

Copy 类型 vs Move 类型

#![allow(unused)]
fn main() {
// Copy 类型（类似 C# 值类型）- 复制，不移动
let x = 5;        // i32 实现 Copy
let y = x;        // x 被复制到 y
println!("{}", x); // ✅ 工作：x 仍然有效

// Move 类型（类似 C# 引用类型）- 移动，不复制
let s1 = String::from("hello");  // String 不实现 Copy
let s2 = s1;                     // s1 被移动到 s2
// println!("{}", s1);           // ❌ 错误：s1 不再有效
}

实际示例：交换值

// C# - 简单引用交换
public void SwapLists(ref List<int> a, ref List<int> b)
{
    var temp = a;
    a = b;
    b = temp;
}

#![allow(unused)]
fn main() {
// Rust - 所有权感知交换
fn swap_vectors(a: &mut Vec<i32>, b: &mut Vec<i32>) {
    std::mem::swap(a, b);  // 内置交换函数
}

// 或手动方法
fn manual_swap() {
    let mut a = vec![1, 2, 3];
    let mut b = vec![4, 5, 6];

    let temp = a;  // 移动 a 到 temp
    a = b;         // 移动 b 到 a
    b = temp;      // 移动 temp 到 b

    println!("a: {:?}, b: {:?}", a, b);
}
}

借用基础

借用类似于 C# 中获取引用，但带有编译时安全保证。

Rust 借用

// Rust - 使用 & 和 &mut 借用
fn modify_value(value: &mut i32) {  // 可变借用
    *value += 10;
}

fn read_value(value: &i32) {        // 不可变借用
    println!("{}", value);
}

fn main() {
    let mut x = 5;

    read_value(&x);      // 不可变借用
    modify_value(&mut x); // 可变借用

    println!("{}", x);   // x 在这里仍然拥有
}

借用规则（在编译时强制执行！）

#![allow(unused)]
fn main() {
fn borrowing_rules() {
    let mut data = vec![1, 2, 3];

    // 规则 1：多个不可变借用可以
    let r1 = &data;
    let r2 = &data;
    println!("{:?} {:?}", r1, r2);  // ✅ 工作

    // 规则 2：一次只能有一个可变借用
    let r3 = &mut data;
    // let r4 = &mut data;  // ❌ 错误：不能两次可变借用
    // let r5 = &data;      // ❌ 错误：可变借用时不能不可变借用

    r3.push(4);  // 使用可变借用
    // r3 在这里超出作用域

    // 规则 3：可以在之前的借用结束后再次借用
    let r6 = &data;  // ✅ 现在可以
    println!("{:?}", r6);
}
}

C# vs Rust：引用安全

// C# - 可能的运行时错误
public class ReferenceSafety
{
    private List<int> data = new List<int>();

    public List<int> GetData() => data;  // 返回内部数据的引用

    public void UnsafeExample()
    {
        var reference = GetData();

        // 另一个线程可能在这里修改数据！
        Thread.Sleep(1000);

        // reference 可能无效或已更改
        reference.Add(42);  // 可能的数据竞争
    }
}

#![allow(unused)]
fn main() {
// Rust - 编译时安全
pub struct SafeContainer {
    data: Vec<i32>,
}

impl SafeContainer {
    // 返回不可变借用 - 调用者不能修改
    // 更喜欢 &[i32] 而不是 &Vec<i32> —— 接受最宽的类型
    pub fn get_data(&self) -> &[i32] {
        &self.data
    }

    // 返回可变借用 - 保证独占访问
    pub fn get_data_mut(&mut self) -> &mut Vec<i32> {
        &mut self.data
    }
}

fn safe_example() {
    let mut container = SafeContainer { data: vec![1, 2, 3] };

    let reference = container.get_data();
    // container.get_data_mut();  // ❌ 错误：不可变借用时不能可变借用

    println!("{:?}", reference);  // 使用不可变引用
    // reference 在这里超出作用域

    let mut_reference = container.get_data_mut();  // ✅ 现在可以
    mut_reference.push(4);
}
}

移动语义

C# 值类型 vs 引用类型

// C# - 值类型被复制
struct Point
{
    public int X { get; set; }
    public int Y { get; set; }
}

var p1 = new Point { X = 1, Y = 2 };
var p2 = p1;  // 复制
p2.X = 10;
Console.WriteLine(p1.X);  // 仍然是 1

// C# - 引用类型共享对象
var list1 = new List<int> { 1, 2, 3 };
var list2 = list1;  // 引用复制
list2.Add(4);
Console.WriteLine(list1.Count);  // 4 - 同一个对象

Rust 移动语义

#![allow(unused)]
fn main() {
// Rust - 默认对非 Copy 类型移动
#[derive(Debug)]
struct Point {
    x: i32,
    y: i32,
}

fn move_example() {
    let p1 = Point { x: 1, y: 2 };
    let p2 = p1;  // 移动（不是复制）
    // println!("{:?}", p1);  // ❌ 错误：p1 被移动了
    println!("{:?}", p2);    // ✅ 工作
}

// 要启用复制，实现 Copy trait
#[derive(Debug, Copy, Clone)]
struct CopyablePoint {
    x: i32,
    y: i32,
}

fn copy_example() {
    let p1 = CopyablePoint { x: 1, y: 2 };
    let p2 = p1;  // 复制（因为它实现 Copy）
    println!("{:?}", p1);  // ✅ 工作
    println!("{:?}", p2);  // ✅ 工作
}
}

何时值被移动

#![allow(unused)]
fn main() {
fn demonstrate_moves() {
    let s = String::from("hello");

    // 1. 赋值移动
    let s2 = s;  // s 移动到 s2

    // 2. 函数调用移动
    take_ownership(s2);  // s2 移动到函数中

    // 3. 从函数返回移动
    let s3 = give_ownership();  // 返回值移动到 s3

    println!("{}", s3);  // s3 有效
}

fn take_ownership(s: String) {
    println!("{}", s);
    // s 在这里被丢弃
}

fn give_ownership() -> String {
    String::from("yours")  // 所有权移动到调用者
}
}

使用借用避免移动

#![allow(unused)]
fn main() {
fn demonstrate_borrowing() {
    let s = String::from("hello");

    // 借用而不是移动
    let len = calculate_length(&s);  // s 被借用
    println!("'{}' has length {}", s, len);  // s 仍然有效
}

fn calculate_length(s: &String) -> usize {
    s.len()  // s 不被拥有，所以不被丢弃
}
}

内存管理：GC vs RAII

C# 垃圾回收

// C# - 自动内存管理
public class Person
{
    public string Name { get; set; }
    public List<string> Hobbies { get; set; } = new List<string>();

    public void AddHobby(string hobby)
    {
        Hobbies.Add(hobby);  // 自动分配内存
    }

    // 不需要显式清理 - GC 处理它
    // 但 IDisposable 模式用于资源
}

using var file = new FileStream("data.txt", FileMode.Open);
// 'using' 确保调用 Dispose()

Rust 所有权和 RAII

#![allow(unused)]
fn main() {
// Rust - 编译时内存管理
pub struct Person {
    name: String,
    hobbies: Vec<String>,
}

impl Person {
    pub fn add_hobby(&mut self, hobby: String) {
        self.hobbies.push(hobby);  // 编译时跟踪内存管理
    }

    // Drop trait 自动实现 - 清理是保证的
    // 与 C# 的 IDisposable 比较：
    //   C#:   using var file = new FileStream(...)    // 在 using 块结束时调用 Dispose()
    //   Rust: let file = File::open(...)?             // 在作用域结束时调用 drop() —— 不需要 'using'
}

// RAII - 资源获取即初始化
{
    let file = std::fs::File::open("data.txt")?;
    // 当 'file' 超出作用域时文件自动关闭
    // 不需要 'using' 语句 - 由类型系统处理
}
}

练习

🏋️ 练习：修复借用检查器错误（点击展开）

挑战：每个片段都有一个借用检查器错误。修复它们而不改变输出。

#![allow(unused)]
fn main() {
// 1. 移动后使用
fn problem_1() {
    let name = String::from("Alice");
    let greeting = format!("Hello, {name}!");
    let upper = name.to_uppercase();  // 提示：借用而不是移动
    println!("{greeting} — {upper}");
}

// 2. 可变 + 不可变借用重叠
fn problem_2() {
    let mut numbers = vec![1, 2, 3];
    let first = &numbers[0];
    numbers.push(4);            // 提示：重新排序操作
    println!("first = {first}");
}

// 3. 返回对局部的引用
fn problem_3() -> String {
    let s = String::from("hello");
    s   // 提示：返回_owned_值，不是 &str
}
}
🔑 解决方案 
#![allow(unused)]
fn main() {
// 1. format! 已经借用——修复是 format! 获取引用。
//    原始代码实际上编译！但如果我们有 `let greeting = name;`
//    那么通过使用 &name 修复：
fn solution_1() {
    let name = String::from("Alice");
    let greeting = format!("Hello, {}!", &name); // 借用
    let upper = name.to_uppercase();             // name 仍然有效
    println!("{greeting} — {upper}");
}

// 2. 在可变操作之前使用不可变借用：
fn solution_2() {
    let mut numbers = vec![1, 2, 3];
    let first = numbers[0]; // 复制 i32 值（i32 是 Copy）
    numbers.push(4);
    println!("first = {first}");
}

// 3. 返回_owned_ String（已经正确——常见的初学者困惑）：
fn solution_3() -> String {
    let s = String::from("hello");
    s // 所有权转移给调用者——这是正确的模式
}
}

关键要点：

  • format!() 借用它的参数——它不移动它们
  • 像 i32 这样的原始类型实现 Copy，所以索引复制值
  • 返回_owned_值将所有权转移给调用者——无生命周期问题
