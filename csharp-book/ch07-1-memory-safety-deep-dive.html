Rust for C# Programmers

引用 vs 指针

你将学到： Rust 引用 vs C# 指针和 unsafe 上下文，生命周期基础， 以及为什么编译时安全证明比 C# 的运行时检查更强（边界检查、空守卫）。

难度： 🟡 中级

C# 指针（Unsafe 上下文）

// C# unsafe 指针（很少使用）
unsafe void UnsafeExample()
{
    int value = 42;
    int* ptr = &value;  // 指向值的指针
    *ptr = 100;         // 解引用并修改
    Console.WriteLine(value);  // 100
}

Rust 引用（默认安全）

#![allow(unused)]
fn main() {
// Rust 引用（总是安全）
fn safe_example() {
    let mut value = 42;
    let ptr = &mut value;  // 可变引用
    *ptr = 100;           // 解引用并修改
    println!("{}", value); // 100
}

// 不需要 "unsafe" 关键字 - 借用检查器确保安全性
}

C# 开发者的生命周期基础

// C# - 可能返回变得无效的引用
public class LifetimeIssues
{
    public string GetFirstWord(string input)
    {
        return input.Split(' ')[0];  // 返回新字符串（安全）
    }

    public unsafe char* GetFirstChar(string input)
    {
        // 这会很危险——返回托管内存的指针
        fixed (char* ptr = input)
            return ptr;  // ❌ 糟糕：方法结束后 ptr 变得无效
    }
}

#![allow(unused)]
fn main() {
// Rust - 生命周期检查防止悬空引用
fn get_first_word(input: &str) -> &str {
    input.split_whitespace().next().unwrap_or("")
    // ✅ 安全：返回的引用与输入有相同的生命周期
}

fn invalid_reference() -> &str {
    let temp = String::from("hello");
    &temp  // ❌ 编译错误：temp 活得不够长
    // temp 会在函数结束时被丢弃
}

fn valid_reference() -> String {
    let temp = String::from("hello");
    temp  // ✅ 工作：所有权转移给调用者
}
}

内存安全：运行时检查 vs 编译时证明

C# - 运行时安全网

// C# 依赖运行时检查和 GC
public class Buffer
{
    private byte[] data;

    public Buffer(int size)
    {
        data = new byte[size];
    }

    public void ProcessData(int index)
    {
        // 运行时边界检查
        if (index >= data.Length)
            throw new IndexOutOfRangeException();

        data[index] = 42;  // 安全，但在运行时检查
    }

    // 带有事件/静态引用的内存泄漏仍然可能
    public static event Action<string> GlobalEvent;

    public void Subscribe()
    {
        GlobalEvent += HandleEvent;  // 可能造成内存泄漏
        // 忘记取消订阅 - 对象不会被收集
    }

    private void HandleEvent(string message) { /* ... */ }

    // 空引用异常仍然可能
    public void ProcessUser(User user)
    {
        Console.WriteLine(user.Name.ToUpper());  // 如果 user.Name 为 null 则为 NullReferenceException
    }

    // 数组访问可能在运行时失败
    public int GetValue(int[] array, int index)
    {
        return array[index];  // 可能的 IndexOutOfRangeException
    }
}

Rust - 编译时保证

#![allow(unused)]
fn main() {
struct Buffer {
    data: Vec<u8>,
}

impl Buffer {
    fn new(size: usize) -> Self {
        Buffer {
            data: vec![0; size],
        }
    }

    fn process_data(&mut self, index: usize) {
        // 当证明安全时，边界检查可以被编译器优化掉
        if let Some(item) = self.data.get_mut(index) {
            *item = 42;  // 安全访问，编译时证明
        }
        // 或使用带显式边界检查的索引：
        // self.data[index] = 42;  // 在 debug 中 panic，但内存安全
    }

    // 内存泄漏不可能 - 所有权系统防止它们
    fn process_with_closure<F>(&mut self, processor: F)
    where F: FnOnce(&mut Vec<u8>)
    {
        processor(&mut self.data);
        // 当 processor 超出作用域时，它自动被清理
        // 无法创建悬空引用或内存泄漏
    }

    // 空指针解引用不可能 - 没有空指针！
    fn process_user(&self, user: &User) {
        println!("{}", user.name.to_uppercase());  // user.name 不能为 null
    }

    // 数组访问是边界检查的或显式 unsafe
    fn get_value(array: &[i32], index: usize) -> Option<i32> {
        array.get(index).copied()  // 如果越界则返回 None
    }

    // 或者如果你知道你在做什么则显式 unsafe：
    /// # 安全
    /// `index` 必须小于 `array.len()`。
    unsafe fn get_value_unchecked(array: &[i32], index: usize) -> i32 {
        *array.get_unchecked(index)  // 快但必须手动证明边界
    }
}

struct User {
    name: String,  // String 在 Rust 中不能为 null
}

// 所有权防止使用后释放
fn ownership_example() {
    let data = vec![1, 2, 3, 4, 5];
    let reference = &data[0];  // 借用 data

    // drop(data);  // 错误：借用时不能丢弃
    println!("{}", reference);  // 这保证安全
}

// 借用防止数据竞争
fn borrowing_example(data: &mut Vec<i32>) {
    let first = &data[0];  // 不可变借用
    // data.push(6);  // 错误：不可变借用时不能可变借用
    println!("{}", first);  // 保证无数据竞争
}
}

练习

🏋️ 练习：发现安全 Bug（点击展开）

这段 C# 代码有一个微妙的安全 bug。识别它，然后编写 Rust 等价物并解释为什么 Rust 版本不会编译：

public List<int> GetEvenNumbers(List<int> numbers)
{
    var result = new List<int>();
    foreach (var n in numbers)
    {
        if (n % 2 == 0)
        {
            result.Add(n);
            numbers.Remove(n);  // Bug：迭代时修改集合
        }
    }
    return result;
}
🔑 解决方案

C# bug：迭代时修改 numbers 在运行时抛出 InvalidOperationException。在代码审查中容易遗漏。

fn get_even_numbers(numbers: &mut Vec<i32>) -> Vec<i32> {
    let mut result = Vec::new();
    for &n in numbers.iter() {
        if n % 2 == 0 {
            result.push(n);
            // numbers.retain(|&x| x != n);
            // ❌ 错误：无法可变借用 `*numbers` 因为
            //    它也被不可变借用（由迭代器）
        }
    }
    result
}

// 惯用 Rust：使用 partition 或 retain
fn get_even_numbers_idiomatic(numbers: &mut Vec<i32>) -> Vec<i32> {
    let evens: Vec<i32> = numbers.iter().copied().filter(|n| n % 2 == 0).collect();
    numbers.retain(|n| n % 2 != 0); // 迭代后移除偶数
    evens
}

fn main() {
    let mut nums = vec![1, 2, 3, 4, 5, 6];
    let evens = get_even_numbers_idiomatic(&mut nums);
    assert_eq!(evens, vec![2, 4, 6]);
    assert_eq!(nums, vec![1, 3, 5]);
}

关键见解：Rust 的借用检查器在编译时防止整个"迭代时变异"bug 类别。C# 在运行时捕获它；许多语言根本不捕获它。
