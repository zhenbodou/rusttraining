Rust for C/C++ Programmers

C++ → Rust 语义深入探讨

你将学到： C++ 概念没有明显 Rust 等效的详细映射——四个命名转换、SFINAE vs trait 约束、CRTP vs 关联类型，以及其他翻译期间的常见摩擦点。

以下部分映射没有明显 1:1 Rust 等效的 C++ 概念。这些差异在 翻译工作期间经常困扰 C++ 程序员。

转换层次结构：四个 C++ 转换 → Rust 等效

C++ 有四个命名转换。Rust 用不同、更显式的机制替换它们：

// C++ 转换层次结构
int i = static_cast<int>(3.14);            // 1. 数字/向上转换
Derived* d = dynamic_cast<Derived*>(base); // 2. 运行时向下转换
int* p = const_cast<int*>(cp);              // 3. 转换掉 const
auto* raw = reinterpret_cast<char*>(&obj); // 4. 位级重新解释
C++ 转换Rust 等效安全说明
static_cast（数字）as 关键字安全但可能截断/回绕let i = 3.14_f64 as i32;——截断为 3
static_cast（数字，检查）From/Into安全，编译时验证let i: i32 = 42_u8.into();——仅扩展
static_cast（数字，可失败）TryFrom/TryInto安全，返回 Resultlet i: u8 = 300_u16.try_into()?;——返回 Err
dynamic_cast（向下转换）match 在 enum 上 / Any::downcast_ref安全enum 的模式匹配；trait 对象的 Any
const_cast无等效Rust 在安全代码中无法将 & 转换为 &mut。对内部可变性使用 Cell/RefCell
reinterpret_caststd::mem::transmuteunsafe重新解释位模式。几乎总是错误——优先使用 from_le_bytes() 等
#![allow(unused)]
fn main() {
// Rust 等效：

// 1. 数字转换——优先 From/Into 而不是 `as`
let widened: u32 = 42_u8.into();             // 不可失败的扩展——始终优先
let truncated = 300_u16 as u8;                // ⚠ 回绕为 44！静默数据丢失
let checked: Result<u8, _> = 300_u16.try_into(); // Err——安全可失败转换

// 2. 向下转换：enum（优先）或 Any（需要类型擦除时）
use std::any::Any;

fn handle_any(val: &dyn Any) {
    if let Some(s) = val.downcast_ref::<String>() {
        println!("获得字符串：{s}");
    } else if let Some(n) = val.downcast_ref::<i32>() {
        println!("获得整数：{n}");
    }
}

// 3. "const_cast" → 内部可变性（无需 unsafe）
use std::cell::Cell;
struct Sensor {
    read_count: Cell<u32>,  // 通过 &self 可变
}
impl Sensor {
    fn read(&self) -> f64 {
        self.read_count.set(self.read_count.get() + 1); // &self，不是 &mut self
        42.0
    }
}

// 4. reinterpret_cast → transmute（几乎从不需要）
// 优先安全替代方案：
let bytes: [u8; 4] = 0x12345678_u32.to_ne_bytes();  // ✅ 安全
let val = u32::from_ne_bytes(bytes);                   // ✅ 安全
// unsafe { std::mem::transmute::<u32, [u8; 4]>(val) } // ❌ 避免
}

指南：在惯用 Rust 中，as 应该罕见（对扩展使用 From/Into， 对缩小使用 TryFrom/TryInto），transmute 应该例外， const_cast 没有等效因为内部可变性 类型使其不必要。

预处理器 → cfg、功能标志和 macro_rules!

C++ 严重依赖预处理器进行条件编译、常量和 代码生成。Rust 用一流语言功能替换所有这些。

#define 常量 → const 或 const fn

// C++
#define MAX_RETRIES 5
#define BUFFER_SIZE (1024 * 64)
#define SQUARE(x) ((x) * (x))  // 宏——文本替换，无类型安全

#![allow(unused)]
fn main() {
// Rust——类型安全、作用域、无文本替换
const MAX_RETRIES: u32 = 5;
const BUFFER_SIZE: usize = 1024 * 64;
const fn square(x: u32) -> u32 { x * x }  // 编译时求值

// 可用于 const 上下文：
const AREA: u32 = square(12);  // 编译时计算
static BUFFER: [u8; BUFFER_SIZE] = [0; BUFFER_SIZE];
}

#ifdef / #if → #[cfg()] 和 cfg!()

// C++
#ifdef DEBUG

#endif

#if defined(LINUX) && !defined(ARM)
    use_x86_path();
#else
    use_generic_path();
#endif

#![allow(unused)]

// Rust——基于属性的条件编译
#[cfg(debug_assertions)]
fn log_verbose(msg: &str) { eprintln!("[VERBOSE] {msg}"); }

#[cfg(not(debug_assertions))]

// 组合条件：
#[cfg(all(target_os = "linux", target_arch = "x86_64"))]
fn use_x86_path() { /* ... */ }

#[cfg(not(all(target_os = "linux", target_arch = "x86_64")))]
fn use_generic_path() { /* ... */ }

// 运行时检查（条件仍然是编译时，但可用于表达式）：
if cfg!(target_os = "windows") {
    println!("在 Windows 上运行");
}
}

#define MACRO(x) → macro_rules!

// C++——文本替换， notoriously 容易出错

    do { if (!(cond)) { log_error(msg); return false; } } while(0)

        if !($cond) {

}

    diag_check!(voltage > 0.8, "Rail 电压太低");
    diag_check!(voltage < 1.5, "Rail 电压太高");
    println!("所有检查通过");
    Ok(())
}
}

头文件和 #include → 模块和 use

在 C++ 中，编译模型围绕文本包含：

// widget.h——每个使用 Widget 的翻译单元包含这个
#pragma once
#include <string>
#include <vector>

class Widget {
public:
    Widget(std::string name);
    void activate();
private:
    std::string name_;
    std::vector<int> data_;
};

// widget.cpp——单独定义
#include "widget.h"
Widget::Widget(std::string name) : name_(std::move(name)) {}
void Widget::activate() { /* ... */ }

在 Rust 中，无头文件、无前向声明、无包含保护：

#![allow(unused)]
fn main() {
// src/widget.rs——声明和定义在一个文件中
pub struct Widget {
    name: String,         // 默认私有
    data: Vec<i32>,
}

impl Widget {
    pub fn new(name: String) -> Self {
        Widget { name, data: Vec::new() }
    }
    pub fn activate(&self) { /* ... */ }
}
}
// src/main.rs——按模块路径导入
mod widget;  // 告诉编译器包含 src/widget.rs
use widget::Widget;

fn main() {
    let w = Widget::new("sensor".to_string());
    w.activate();
}

#include "foo.h"父级中的 mod foo; + use foo::Item;无文本包含，无 ODR 违规
#pragma once / 包含保护不需要每个 .rs 文件是模块——编译一次

class Foo;（不完整类型）不需要无单独声明/定义拆分
每个类的 .h + .cpp单个 .rs 文件无声明/定义不匹配 bug
using namespace std;use std::collections::HashMap;始终显式——无全局命名空间污染
嵌套 namespace a::b嵌套 mod a { mod b { } } 或 a/b.rs文件系统镜像模块树

friend 和访问控制 → 模块可见性

C++ 使用 friend 授予特定类或函数访问私有成员的权限。 Rust 无 friend 关键字——相反，可见性是模块作用域：

// C++
class Engine {
    friend class Car;   // Car 可以访问私有成员
    int rpm_;
    void set_rpm(int r) { rpm_ = r; }
public:
    int rpm() const { return rpm_; }
};

// Rust——同一模块中的项可以访问所有字段，无需 `friend`
mod vehicle {
    pub struct Engine {
        rpm: u32,  // 对模块私有（不是对结构！）
    }

    impl Engine {
        pub fn new() -> Self { Engine { rpm: 0 } }
        pub fn rpm(&self) -> u32 { self.rpm }
    }

    pub struct Car {
        engine: Engine,
    }

    impl Car {
        pub fn new() -> Self { Car { engine: Engine::new() } }
        pub fn accelerate(&mut self) {
            self.engine.rpm = 3000; // ✅ 同一模块——直接字段访问
        }
        pub fn rpm(&self) -> u32 {
            self.engine.rpm  // ✅ 同一模块——可以读取私有字段
        }
    }
}

fn main() {
    let mut car = vehicle::Car::new();
    car.accelerate();
    // car.engine.rpm = 9000;  // ❌ 编译错误：`engine` 是私有的
    println!("RPM: {}", car.rpm()); // ✅ Car 上的公共方法
}
C++ 访问Rust 等效作用域
private（默认，无关键字）仅在同一模块内可访问
protected无直接等效对父模块使用 pub(super)
publicpub到处可访问
friend class Foo将 Foo 放在同一模块中模块级可见性替换 friend
—pub(crate)在 crate 内可见但对外部依赖者不可见
—pub(super)仅对父模块可见
—pub(in crate::path)在特定模块子树内可见

关键见解：C++ 可见性是每类的。Rust 可见性是每模块的。 这意味着你通过选择哪些类型生活在同一模块中来控制访问—— 共置类型可以完全访问彼此的私有字段。

volatile → 原子操作和 read_volatile/write_volatile

在 C++ 中，volatile 告诉编译器不要优化掉读/写——通常 用于内存映射硬件寄存器。Rust 无 volatile 关键字。

// C++：用于硬件寄存器的 volatile
volatile uint32_t* const GPIO_REG = reinterpret_cast<volatile uint32_t*>(0x4002'0000);
*GPIO_REG = 0x01;              // 写入不被优化掉
uint32_t val = *GPIO_REG;     // 读取不被优化掉

#![allow(unused)]
fn main() {
// Rust：显式易失操作——仅在 unsafe 代码中
use std::ptr;

const GPIO_REG: *mut u32 = 0x4002_0000 as *mut u32;

// 安全：GPIO_REG 是有效的内存映射 I/O 地址。
unsafe {
    ptr::write_volatile(GPIO_REG, 0x01);   // 写入不被优化掉
    let val = ptr::read_volatile(GPIO_REG); // 读取不被优化掉
}
}

对于并发共享状态（另一个常见的 C++ volatile 用例），Rust 使用原子操作：

// C++：volatile 对线程安全不足（常见错误！）
volatile bool stop_flag = false;  // ❌ 数据竞争——C++11+ 中的 UB

// 正确的 C++：
std::atomic<bool> stop_flag{false};

#![allow(unused)]
fn main() {
// Rust：原子操作是跨线程共享可变状态的唯一方式
use std::sync::atomic::{AtomicBool, Ordering};

static STOP_FLAG: AtomicBool = AtomicBool::new(false);

// 从另一个线程：
STOP_FLAG.store(true, Ordering::Release);

// 检查：
if STOP_FLAG.load(Ordering::Acquire) {
    println!("停止");
}
}
C++ 用例Rust 等效说明
用于硬件寄存器的 volatileptr::read_volatile / ptr::write_volatile需要 unsafe——对 MMIO 正确
用于线程信令的 volatileAtomicBool / AtomicU32 等C++ volatile 对此也错误！
std::atomic<T>std::sync::atomic::AtomicT相同语义，相同顺序
std::atomic<T>::load(memory_order_acquire)AtomicT::load(Ordering::Acquire)1:1 映射

static 变量 → static、const、LazyLock、OnceLock

基本 static 和 const

// C++
const int MAX_RETRIES = 5;                    // 编译时常量
static std::string CONFIG_PATH = "/etc/app";  // 静态初始化——顺序未定义！

#![allow(unused)]
fn main() {
// Rust
const MAX_RETRIES: u32 = 5;                   // 编译时常量，内联
static CONFIG_PATH: &str = "/etc/app";         // 'static 生命周期，固定地址
}

静态初始化顺序困境

C++ 有一个知名问题：不同翻译单元中的全局构造函数 以未指定顺序执行。Rust 完全避免这个——static 值必须是 编译时常量（无构造函数）。

对于运行时初始化的全局变量，使用 LazyLock（Rust 1.80+）或 OnceLock：

#![allow(unused)]
fn main() {
use std::sync::LazyLock;

// 等效于 C++ `static std::regex`——首次访问时初始化，线程安全
static CONFIG_REGEX: LazyLock<regex::Regex> = LazyLock::new(|| {
    regex::Regex::new(r"^[a-z]+_diag$").expect("无效的正则表达式")
});

fn is_valid_diag(name: &str) -> bool {
    CONFIG_REGEX.is_match(name)  // 首次调用初始化；后续调用快速
}
}
#![allow(unused)]
fn main() {
use std::sync::OnceLock;

// OnceLock：初始化一次，可以从运行时数据设置
static DB_CONN: OnceLock<String> = OnceLock::new();

fn init_db(connection_string: &str) {
    DB_CONN.set(connection_string.to_string())
        .expect("DB_CONN 已初始化");
}

fn get_db() -> &'static str {
    DB_CONN.get().expect("DB 未初始化")
}
}

const int X = 5;const X: i32 = 5;两者编译时。Rust 需要类型注解
constexpr int X = 5;const X: i32 = 5;Rust const 始终是 constexpr
static int count = 0;（文件作用域）static COUNT: AtomicI32 = AtomicI32::new(0);可变静态需要 unsafe 或原子操作
static std::string s = "hi";static S: &str = "hi"; 或 LazyLock<String>简单情况无运行时构造函数
static MyObj obj;（复杂初始化）static OBJ: LazyLock<MyObj> = LazyLock::new(|| { ... });线程安全、惰性、无初始化顺序问题
thread_localthread_local! { static X: Cell<u32> = Cell::new(0); }相同语义

constexpr → const fn

C++ constexpr 标记函数和变量用于编译时求值。Rust 对相同目的使用 const fn 和 const：

// C++
constexpr int factorial(int n) {
    return n <= 1 ? 1 : n * factorial(n - 1);
}
constexpr int val = factorial(5);  // 编译时计算 → 120

#![allow(unused)]
fn main() {
// Rust
const fn factorial(n: u32) -> u32 {
    if n <= 1 { 1 } else { n * factorial(n - 1) }
}
const VAL: u32 = factorial(5);  // 编译时计算 → 120

// 也可用于数组大小和 match 模式：
const LOOKUP: [u32; 5] = [factorial(1), factorial(2), factorial(3),
                           factorial(4), factorial(5)];
}
C++Rust说明
constexpr int f()const fn f() -> i32相同意图——编译时可求值
constexpr 变量const 变量Rust const 始终是编译时
consteval（C++20）无等效const fn 也可以在运行时运行
if constexpr（C++17）无等效（使用 cfg! 或泛型）trait 专门化填充一些用例
constinit（C++20）带 const 初始化器的 staticRust static 默认必须 const 初始化

const fn 的当前限制（自 Rust 1.82 起稳定）：

  • 无 trait 方法（不能在 const 上下文中调用 Vec 的 .len()）
  • 无堆分配（Box::new、Vec::new 不是 const）
  • 无浮点算术——在 Rust 1.82 中稳定
  • 不能使用 for 循环（使用递归或带手动索引的 while）

SFINAE 和 enable_if → Trait 约束和 where 子句

在 C++ 中，SFINAE（替换失败不是错误）是条件泛型编程背后的机制。它功能强大但以难以阅读著称。Rust 用trait 约束完全替换它：

// C++：基于 SFINAE 的条件函数（C++20 之前）
template<typename T,
         std::enable_if_t<std::is_integral_v<T>, int> = 0>
T double_it(T val) { return val * 2; }

template<typename T,
         std::enable_if_t<std::is_floating_point_v<T>, int> = 0>
T double_it(T val) { return val * 2.0; }

// C++20 concepts——更清晰但仍然冗长：
template<std::integral T>
T double_it(T val) { return val * 2; }

#![allow(unused)]
fn main() {
// Rust：trait 约束——可读、可组合、出色的错误消息
use std::ops::Mul;

fn double_it<T: Mul<Output = T> + From<u8>>(val: T) -> T {
    val * T::from(2)
}

// 或使用 where 子句用于复杂约束：
fn process<T>(val: T) -> String
where
    T: std::fmt::Display + Clone + Send,
{
    format!("处理：{}", val)
}

// 通过单独的 impl 的条件行为（替换 SFINAE 重载）：
trait Describable {
    fn describe(&self) -> String;
}

impl Describable for u32 {
    fn describe(&self) -> String { format!("整数：{self}") }
}

impl Describable for f64 {
    fn describe(&self) -> String { format!("浮点：{self:.2}") }
}
}

std::is_integral_v<T>在数字 trait 或特定类型上的约束🟢 无 _v / _t 后缀

编译失败在模板深处编译失败在调用点，trait 不匹配🟢 无 200 行错误级联

关键见解：C++ concepts（C++20）最接近 Rust trait。 如果你熟悉 C++20 concepts，将 Rust trait 视为 concepts 自 1.0 以来是一流语言功能，具有连贯的 实现模型（trait impl）而不是鸭子类型。

std::function → 函数指针、impl Fn 和 Box<dyn Fn>

C++ std::function<R(Args...)> 是类型擦除的可调用对象。Rust 有三个选项， 每个都有不同的权衡：

// C++：多合一（堆分配，类型擦除）
#include <functional>
std::function<int(int)> make_adder(int n) {
    return [n](int x) { return x + n; };
}

#![allow(unused)]
fn main() {
// Rust 选项 1：fn 指针——简单，无捕获，无分配
fn add_one(x: i32) -> i32 { x + 1 }
let f: fn(i32) -> i32 = add_one;
println!("{}", f(5)); // 6

// Rust 选项 2：impl Fn——单态化，零开销，可以捕获
fn apply(val: i32, f: impl Fn(i32) -> i32) -> i32 { f(val) }
let n = 10;
let result = apply(5, |x| x + n);  // 闭包捕获 `n`

// Rust 选项 3：Box<dyn Fn>——类型擦除，堆分配（如 std::function）
fn make_adder(n: i32) -> Box<dyn Fn(i32) -> i32> {
    Box::new(move |x| x + n)
}
let adder = make_adder(10);
println!("{}", adder(5));  // 15

// 存储异构可调用对象（如 vector<function<int(int)>>）：
let callbacks: Vec<Box<dyn Fn(i32) -> i32>> = vec![
    Box::new(|x| x + 1),
    Box::new(|x| x * 2),
    Box::new(make_adder(100)),
];
for cb in &callbacks {
    println!("{}", cb(5));  // 6, 10, 105
}
}
何时使用C++ 等效Rust 选择
顶层函数，无捕获函数指针fn(Args) -> Ret
接受可调用对象的泛型函数模板参数impl Fn(Args) -> Ret（静态分发）
泛型中的 trait 约束template<typename F>F: Fn(Args) -> Ret
存储的可调用对象，类型擦除std::function<R(Args)>Box<dyn Fn(Args) -> Ret>
突变状态的回调带可变 lambda 的 std::functionBox<dyn FnMut(Args) -> Ret>
一次性回调（消耗）std::function（移动）Box<dyn FnOnce(Args) -> Ret>

性能说明：impl Fn 零开销（单态化，如 C++ 模板）。 Box<dyn Fn> 与 std::function 相同开销（vtable + 堆分配）。 除非你需要存储异构可调用对象，否则优先使用 impl Fn。

容器映射：C++ STL → Rust std::collections

C++ STL 容器Rust 等效说明
std::vector<T>Vec<T>几乎相同的 API。Rust 默认检查边界
std::array<T, N>[T; N]栈分配的固定大小数组
std::deque<T>std::collections::VecDeque<T>环形缓冲区。两端高效 push/pop
std::list<T>std::collections::LinkedList<T>Rust 中很少使用——Vec 几乎总是更快
std::forward_list<T>无等效使用 Vec 或 VecDeque
std::unordered_map<K, V>std::collections::HashMap<K, V>默认使用 SipHash（抗 DoS）
std::map<K, V>std::collections::BTreeMap<K, V>B 树；键排序；需要 K: Ord
std::unordered_set<T>std::collections::HashSet<T>需要 T: Hash + Eq
std::set<T>std::collections::BTreeSet<T>排序集合；需要 T: Ord
std::priority_queue<T>std::collections::BinaryHeap<T>默认最大堆（与 C++ 相同）
std::stack<T>Vec<T> 带 .push() / .pop()无需单独的栈类型
std::queue<T>VecDeque<T> 带 .push_back() / .pop_front()无需单独的队列类型
std::stringString保证 UTF-8，非 null 终止
std::string_view&str借用的 UTF-8 切片
std::span<T>（C++20）&[T] / &mut [T]Rust 切片自 1.0 以来是一流类型
std::tuple<A, B, C>(A, B, C)一流语法，可解构
std::pair<A, B>(A, B)只是 2 元素元组
std::bitset<N>无 std 等效使用 bitvec crate 或 [u8; N/8]

关键区别：

  • Rust 的 HashMap/HashSet 需要 K: Hash + Eq——编译器在类型级别强制执行这个，不像 C++ 中使用不可哈希键会在 STL 深处给出模板错误
  • Vec 索引（v[i]）默认在越界时 panic。使用 .get(i) 获取 Option<&T> 或迭代器完全避免边界检查
  • 无 std::multimap 或 std::multiset——使用 HashMap<K, Vec<V>> 或 BTreeMap<K, Vec<V>>

异常安全 → Panic 安全

C++ 定义三个级别的异常安全（Abrahams 保证）：

C++ 级别含义Rust 等效
无抛出函数从不抛出函数从不 panic（返回 Result）
强（提交或回滚）如果抛出，状态不变所有权模型使这自然——如果 ? 提前返回，部分构建的值被删除
基本如果抛出，不变量保留Rust 的默认——Drop 运行，无泄漏

Rust 的所有权模型如何帮助

#![allow(unused)]
fn main() {
// 免费强保证——如果 file.write() 失败，config 不变
fn update_config(config: &mut Config, path: &str) -> Result<(), Error> {
    let new_data = fetch_from_network()?; // Err → 提前返回，config 未触及
    let validated = validate(new_data)?;   // Err → 提前返回，config 未触及
    *config = validated;                   // 仅在成功时到达（提交）
    Ok(())
}
}

在 C++ 中，实现强保证需要手动回滚或复制交换 惯用法。在 Rust 中，? 传播为大多数代码默认给你强保证。

catch_unwind——Rust 的 catch(...) 等效

#![allow(unused)]
fn main() {
use std::panic;

// 捕获 panic（如 C++ 中的 catch(...)）——很少需要
let result = panic::catch_unwind(|| {
    // 可能 panic 的代码
    let v = vec![1, 2, 3];
    v[10]  // Panic！（索引越界）
});

match result {
    Ok(val) => println!("获得：{val}"),
    Err(_) => eprintln!("捕获 panic——已清理"),
}
}
